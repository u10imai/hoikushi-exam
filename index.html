    let userEmail = "";

    let currentSubject = "";

    let currentQuiz = [];

    let currentIndex = 0;

    let score = 0;

    let resultLog = "";


    function goToSlide(slideId) {

        document.querySelectorAll('.slide-container').forEach(s => s.classList.remove('slide-active'));

        document.getElementById(slideId).classList.add('slide-active');

    }


    function saveUser() {

        const email = document.getElementById('student-email').value;

        if (email.includes('@')) {

            userEmail = email;

            goToSlide('slide-subjects');

        } else {

            alert('正しいメールアドレスを入力してください。');

        }

    }


    function selectSubject(subject) {

        currentSubject = subject;

        if (subject === "MIX") {

            // 全科目の問題を結合してシャッフル

            currentQuiz = Object.values(masterData).flat().sort(() => Math.random() - 0.5);

            document.getElementById('quiz-title').innerText = "演習：全科目MIX";

        } else {

            // 選択した科目の問題をシャッフル

            currentQuiz = [...masterData[subject]].sort(() => Math.random() - 0.5);

            document.getElementById('quiz-title').innerText = `演習：${subject}`;

        }

        

        currentIndex = 0;

        score = 0;

        resultLog = "";

        startQuiz();

    }


    function startQuiz() {

        if (currentQuiz.length === 0) {

            alert("この科目の問題はまだ登録されていません。");

            return;

        }

        goToSlide('slide-quiz');

        loadQuestion();

    }


    function loadQuestion() {

        const q = currentQuiz[currentIndex];

        document.getElementById('question-text').innerText = `Q. ${q.question}`;

        document.getElementById('progress-fill').style.width = `${(currentIndex / currentQuiz.length) * 100}%`;

        

        const container = document.getElementById('options-container');

        container.innerHTML = '';

        

        q.options.forEach((opt, idx) => {

            const btn = document.createElement('button');

            btn.className = 'option-btn';

            btn.innerText = opt;

            btn.onclick = () => handleAnswer(idx, opt);

            container.appendChild(btn);

        });

    }


    function handleAnswer(idx, optText) {

        const q = currentQuiz[currentIndex];

        const isCorrect = (idx === q.answer);

        if (isCorrect) score++;


        resultLog += `【Q${currentIndex + 1}】${isCorrect ? '○' : '×'} 回答: ${optText}\n(正解: ${q.options[q.answer]})\n解説: ${q.explanation}\n\n`;


        currentIndex++;

        if (currentIndex < currentQuiz.length) {

            loadQuestion();

        } else {

            finishQuiz();

        }

    }


    function finishQuiz() {

        document.getElementById('final-score').innerText = score;

        document.getElementById('accuracy').innerText = Math.round((score / currentQuiz.length) * 100);

        goToSlide('slide-result');

    }


    async function sendToGAS() {

        const btn = document.getElementById('submit-btn');

        btn.disabled = true;

        btn.innerText = "送信中...";


        const payload = {

            email: userEmail,

            score: score,

            total: currentQuiz.length,

            details: resultLog

        };


        try {

            await fetch(GAS_URL, {

                method: "POST",

                body: JSON.stringify(payload)

            });

            document.getElementById('status-msg').style.display = 'block';

            btn.style.display = 'none';

        } catch (e) {

            alert("送信に失敗しました。URLを確認してください。");

            btn.disabled = false;

            btn.innerText = "結果を再送する";

        }

    }
